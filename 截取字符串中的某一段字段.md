#### 截取字符串中的某一段字段

```javascript
// 首先是一段字符串
var str = `<add key="FacebookPixelId" value="1351800764881994" />
<add key="CarrefourDeployVersion" value="3.0.144.71.111.28" />
<add key="IsProductionServer" value="False" />`;

// 然后去掉空格
str = str.split(" ").join("");

// 使用 match 来获取相匹配的内容
var version = str.match(/CarrefourDeployVersion"value="(\S*)"/)[1];
console.log(version);  // 3.0.144.71.111.28
```
